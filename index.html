En programmation, les nombres entiers sont aussi appelés entiers ou integers ; les nombres avec des chiffres après la virgule sont aussi appelés nombres en virgule flottante ou floating-point. A class allows you to create objects faster. Here is how to create one : class Book { constructor(title, author, pages) { this.title = title; this.author = author; this.pages = pages; } } Then use new to create classes : let myBook = new Book("L'Histoire de Tao", "Will Alexander", 250); //Cette ligne crée l'objet suivant : { title: "L'Histoire de Tao", author: "Will Alexander", pages: 250 } Arrays let guests = ["Sarah Kate", "Audrey Simon", "Will Alexander"]; Length : let guests = ["Will Alexander", "Sarah Kate", "Audrey Simon"]; let howManyGuests = guests.length; // 3 Push : guests.push("Tao Perkington"); // ajoute "Tao Perkington" à la fin de notre tableau guests Unshift : guests.unshift("Tao Perkington"); // "Tao Perkington" est ajouté au début du tableau guests Pop : guests.pop(); // supprimer le dernier élément du tableau Logical comparisons let userLoggedIn = true; let UserHasPremiumAccount = true; let userHasMegaPremiumAccount = false; userLoggedIn && userHasPremiumAccount; // true userLoggedIn && userHasMegaPremiumAccount; // false userLoggedIn || userHasPremiumAccount; // true userLoggedIn || userHasMegaPremiumAccount; // true !userLoggedIn; // false !userHasMegaPremiumAccount; // true Loops For : const numberOfPassengers = 10; for (let i = 0; i < numberOfPassengers; i++) { console.log("Passager embarqué !"); } For in : const passengers = [ "Will Alexander", "Sarah Kate'", "Audrey Simon", "Tao Perkington" ] for (let i in passengers) { console.log("Embarquement du passager " + passengers[i]); } i starts at 0 and +1 at each loop. This is useful when manipulating a precise i in iteration. For of : const passengers = [ "Will Alexander", "Sarah Kate", "Audrey Simon", "Tao Perkington" ] for (let passenger of passengers) { console.log("Embarquement du passager " + passenger); } If precise i not necessary, it is easier to use for of.
